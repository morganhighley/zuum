Security System Drivers

This topic describes how to create a Crestron Driver for a security system device.

NOTE: Version 3.x of the SDK framework must be used when creating security system drivers. For more information on versioning, refer to SDK Framework Versioning.

Overview

Security systems are represented by the interface ISecuritySystem.

Each supported interface is described in the sections that follow.

ISecuritySystem

ISecuritySystem is base interface to implement the security system. It has information about the area, zone, keypad, commands, alarms, system state, system errors, and capabilities of the security system.

Refer to the sample code below.

Copy 
/// <summary>
/// An interface to define a Security System
/// </summary>
public interface ISecuritySystem  
{
    /// <summary>
    /// Gets the Security System's emulated keypad
    /// </summary>
    /// <returns><see cref=" IEmulatedSecuritySystemKeypad "/></returns>
    IEmulatedSecuritySystemKeypad GetEmulatedKeypad ();

    /// <summary>
    /// Gets a list of all areas in Security system
    /// </summary>
    /// <returns><see cref=" ISecuritySystemArea "/></returns>
    IEnumerable <ISecuritySystemArea> GetAreas ();

    /// <summary>
    /// Gets the Security System's area by given index
    /// </summary>
    /// <param name="index"></param>
    /// <returns><see cref=" ISecuritySystemArea "/></returns>
    ISecuritySystemArea GetArea (int index);

    /// <summary>
    /// Gets a list of all the zones in Security System
    /// </summary>
    /// <returns><see cref=" ISecuritySystemZone "/></returns>
    IEnumerable<ISecuritySystemZone> GetAllZones ();

    /// <summary>
    /// Gets a list of all the available area commands in Security System
    /// </summary>
    /// <returns><see cref="SecuritySystemAreaCommand"/></returns>
    ReadOnlyCollection<SecuritySystemAreaCommand> GetAvailableAreaCommands ();

    /// <summary>
    /// Gets a list of all the available security system alarms
    /// </summary>
    /// <returns><see cref="SecuritySystemAlarmType"/></returns>
    ReadOnlyCollection<SecuritySystemAlarmType> GetAvailableSystemAlarms ();

    /// <summary>
    /// Gets the list of all available system states in Security System
    /// </summary>
    /// <returns><see cref="SecuritySystemState"/></returns>
    ReadOnlyCollection<SecuritySystemState> GetAvailableSystemStates ();

    /// <summary>
    /// Gets the list of currently active security system's alarms
    /// </summary>
    /// <returns><see cref="SecuritySystemAlarmType"/></returns>
    IEnumerable<SecuritySystemAlarmType> GetCurrentSystemAlarms ();

    /// <summary>
    /// Gets the list of currently active security system states
    /// </summary>
    /// <returns><see cref="SecuritySystemState"/></returns>
    IEnumerable<SecuritySystemState> GetCurrentSystemStates ();

    /// <summary>
    /// Gets the list of currently active security system errors
    /// </summary>
    /// <returns><see cref="SecuritySystemError"/></returns>
    IEnumerable<SecuritySystemError> GetActiveErrors ();

    /// <summary>
    /// Gets the list of capabilities in security system
    /// </summary>
    /// <returns><see cref="SecuritySystemCapabilities"/></returns>
    ReadOnlyCollection<SecuritySystemCapabilities> GetCapabilities ();

    /// <summary>
    /// Sends the Security System's area command
    /// </summary>
    /// <param name="areaIndexes">The area index</param>
    /// <param name="commandIndex">The command index</param>
    /// <param name="password">Password for command</param>
    /// <returns><see cref="SecuritySystemOperationalResult"/></returns>
     SecuritySystemOperationalResult SendAreaCommand (List<int> areaIndexes, int commandIndex, string password);

    /// <summary>
    /// Bypass the Security System's zone
    /// </summary>
    /// <param name="areaIndex">The area index</param>
    /// <param name="zoneIndex">The zone index</param>
    /// <param name="password">Password for zone</param>
    void BypassZone (int areaIndex, int zoneIndex, string password);

    /// <summary>
    /// Event gets raised as a result of command being issued to the security system            
    /// <see cref="SecuritySystemCommand"/>
    /// </summary>
    event EventHandler<SecuritySystemCommandResultEventArgs> SecuritySystemCommandResult;

    /// <summary>
    /// Event gets raised as a result of error being issued to the security system
    /// <see cref="SecuritySystemError"/>
    /// </summary>
    event EventHandler<ListChangedEventArgs<SecuritySystemError>> SecuritySystemErrorChanged;
    
     /// <summary>
    /// Event gets raised as a result of alarm state being issued to the  security system
    /// <see cref="SecuritySystemAlarmType"/>
    /// </summary>
    event EventHandler<ListChangedEventArgs<SecuritySystemAlarmType>> SecuritysystemAlarmStateChangedEvent;

    /// <summary>
    /// Event gets raised as a result of state being issued to the security system
    /// <see cref="SecuritySystemState"/>
    /// </summary>
    event EventHandler<ListChangedEventArgs<SecuritySystemState>> SecuritysystemStateChangedEvent;

    /// <summary>
    /// Event gets raised as a result of area list being issued to the security system
    /// <see cref="ISecuritySystemArea"/>
    /// </summary>
    event EventHandler<ListChangedEventArgs<ISecuritySystemArea>> SecuritySystemAreaListChanged;

    /// <summary>
    /// Event gets raised as a result of zone list being issued to the security system
    /// <see cref="ISecuritySystemZone"/>
    /// </summary>
    event EventHandler<ListChangedEventArgs<ISecuritySystemZone>> SecuritySystemZoneListChanged;
}

Create a Sample Driver

The following sections describe how to create a sample security system driver using the included sample classes and protocols.

Create the Driver Project

For information on creating the driver project, refer to Create the Driver Project.

NOTE: In addition to RADCommon, RADProTransports must be added as a reference if creating a serial Security System driver.

Create the Driver Files

The following driver files need to be created for a security system driver. For more information on creating the driver files, refer to Create the Driver Files.

NOTE: If implementing a serial Security System driver, ISerialComport and ISimpl must be implemented instead of ITcp, and a different JSON file must be used.
  • SecuritySystemDriverIP.cs: Implements the ISecuritySystem interface for IP security systems.
  • SecuritySystemDriverSerial.cs: Implements the ISecuritySystem interface for serial security systems.
  • SecuritySystemProtocol.cs: Handles all API interactions with the device.
  • SecuritySystemDriver.json: An embedded resource containing information about the driver, the device, and the API.
  • SecuritySystemKeypad.cs: An implementation of IEmulatedSecuritySystemKeypad interface.
  • SecuritySystemArea.cs: An implementation of ISecuritySystemArea interface.
  • SecuritySystemZone.cs: An implementation of ISecuritySystemZone4 interface.

Implement the Driver Classes

The following topics provide sample implementations for the security system driver classes.

    public class SecuritySystemDriverIP : ABasicDriver,
    ISecuritySystem,
    IAuthentication3,
    IConnection3, ITcp

            try
            {

                    EnableAutoReconnect = EnableAutoReconnect,

                _tcpTransport.DriverID = DriverID;

                    CustomLogger = InternalCustomLogger
                };

                (_securitySystemKeypad as SecuritySystemKeypad).Initialize(_securitySystemProtocol);


                _securitySystemProtocol.RxOut += SendRxOut;
                _securitySystemProtocol.StateChange += OnSecuritySystemStateChanged;
                _securitySystemProtocol.AlarmChange += OnSecurityAlarmChanged;

        public SecuritySystemOperationalResult SendAreaCommand(List<int> areaIndexes, int commandIndex, string password)
        {
        SecuritySystemAreaCommand areaCommand = null;

            if (_availableAreaCommands != null)
            {
                foreach (var command in _availableAreaCommands)
                {
                    if (command.Index == commandIndex)
                    {
                        areaCommand = command;
                        break;
                    }
                }
            }

            SecuritySystemOperationalResult result = null;
            if (areaCommand != null)
            {
                switch (areaCommand.CommandType)
                {
                    case SecuritySystemCommandType.Disarm:
                        result = DisarmAreas(areaIndexes, password);
                        break;
                    default:
                        result = new SecuritySystemOperationalResult(0)
                        {
                            CommandType = SecuritySystemCommandType.Unknown,
                            Result = SecuritySystemOperationalResultCode.Success,
                            TargetComponentId = new List<int>()
                        };
                        break;
                }
            }


            return result;      
          }

                baseRootObject.CrestronSerialDeviceApi.Api.Communication.CommunicationType = obj.CrestronSerialDeviceApi.Api.Communication.CommunicationType;

                baseRootObject.CrestronSerialDeviceApi.Api.Communication.Baud = obj.CrestronSerialDeviceApi.Api.Communication.Baud;
                baseRootObject.CrestronSerialDeviceApi.Api.Communication.Parity = obj.CrestronSerialDeviceApi.Api.Communication.Parity;
                baseRootObject.CrestronSerialDeviceApi.Api.Communication.HwHandshake = obj.CrestronSerialDeviceApi.Api.Communication.HwHandshake;
                baseRootObject.CrestronSerialDeviceApi.Api.Communication.SwHandshake = obj.CrestronSerialDeviceApi.Api.Communication.SwHandshake;

                if (Leds[newLed.Index].State != newLed.State)

            }

            if (FunctionButtons != null)
            {
                if (FunctionButtons[functionButton.FunctionId].Led.State != functionButton.Led.State)

                var alram = new SecuritySystemAlarm(SecuritySystemAlarmType.Unknown, false); var alarmState = new SecuritySystemAlarmState() { Alarm = alram, State = false };
                AlarmChange.Invoke(alarmState);
            }
        }

        private void FireEvent(SecuritySystemStateObjects type, object obj)
        {
            if (StateChange != null)
            {
                StateChange(type, obj);
            }
        }


        #endregion
    }
